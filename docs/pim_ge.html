301
302    Parameters
303    ----------
304    source : SourceLocation
305        Emission source position :math:`(\tilde{x}, \tilde{y}, \tilde{z})`.
306    sensor_positions : Array, shape (N_sensors, 3)
307        Sensor `(x, y, z)` coordinates [m].
308    wind : WindField
309        Per-timestep wind speed/direction, length `T`.
310    mixing_height : float, default 500.0
311        Boundary-layer mixing height `H` [m], used by the inversion-layer
312        reflection term.
313    scheme : {"Briggs", "SMITH", "Draxler"}, default "Briggs"
314        Dispersion parametrization (ignored if `estimated=True`).
315    stability_class : str, default "D"
316        Pasquill-Gifford stability class, used by fixed-table schemes.
317    estimated : bool, default False
318        If True, infer `a_H, b_H, a_V, b_V` from `log_params` instead of a
319        fixed-table scheme.
320    log_params : Array, optional
321        `[log_a_H, log_a_V, log_b_H, log_b_V]`, exponentiated to recover the
322        positive dispersion coefficients (required if `estimated=True`).
323    tan_gamma_H, tan_gamma_V : float, default 1.0
324        Wind-direction roughness terms for the horizontal/vertical power law.
325    source_half_width : float, default 0.0
326        Virtual source offset `w` for the horizontal dispersion term.
327
328    Returns
329    -------
330    Array, shape (T, N_sensors)
331        Coupling matrix `A` [ppm per kg/s], such that `data = A @ s + beta + noise`
332        (Eq. 5). Entries are 0 for sensors upwind of the source (`x_down <= 0`).
333
334    Notes
335    -----
336    Paper Mapping: Newman et al. (2024), Eq. (2), §2.1 — steady-state Gaussian
337    plume solution of the advection-diffusion equation (Eq. 1) with
338    image-source reflections, scaled to ppm via `methane_kg_m3_to_ppm`
339    (the `1e6/rho_CH4` factor in Eq. 2).
340
341    .. math::
342        c = \frac{10^6}{\rho_{CH_4}}\,
343            \frac{s}{2\pi u\,\sigma_H\sigma_V}\,
344            \exp\!\left(-\frac{\delta_H^2}{2\sigma_H^2}\right)
345            \sum_{j} \exp\!\left(-\frac{\delta_{V,j}^2}{2\sigma_V^2}\right)
346
347    The implementation truncates the image-source sum over `j` to four terms:
348    direct (`delta_V`), ground reflection (`delta_V + 2h`), inversion-layer
349    reflection (`delta_V - 2(H - h)`), and one second-order ceiling reflection
350    (`delta_V + 2H`), where `h = source.z` and `H = mixing_height`.
351    """
352    T = wind.speed.shape[0]
353    N = sensor_positions.shape[0]
354
355    sx, sy, sz = sensor_positions[:, 0], sensor_positions[:, 1], sensor_positions[:, 2]
356
357    # (T, N) downwind distances and offsets
358    x_down = (
359        jnp.outer(jnp.ones(T), sx - source.x) * jnp.cos(wind.direction)[:, None]
360        + jnp.outer(jnp.ones(T), sy - source.y) * jnp.sin(wind.direction)[:, None]
361    )
362    y_cross = (
363        -jnp.outer(jnp.ones(T), sx - source.x) * jnp.sin(wind.direction)[:, None]
364        + jnp.outer(jnp.ones(T), sy - source.y) * jnp.cos(wind.direction)[:, None]
365    )
366    dz = sz - source.z  # (N,)
367
368    downwind_mask = (x_down > 0.0).astype(jnp.float32)
369    x_safe = jnp.where(x_down > 0.0, x_down, jnp.ones_like(x_down))
370
371    if estimated and log_params is not None:
372        a_H = jnp.exp(log_params[0])
373        a_V = jnp.exp(log_params[1])
374        b_H = jnp.exp(log_params[2])
375        b_V = jnp.exp(log_params[3])
376        sig_y = horizontal_stddev(
377            x_safe,
378            scheme=scheme,
379            estimated=True,
380            a_H=a_H,
381            b_H=b_H,
382            tan_gamma_H=tan_gamma_H,
383            source_half_width=source_half_width,
384        )
385        sig_z = vertical_stddev(
386            x_safe,
387            scheme=scheme,
388            estimated=True,
389            a_V=a_V,
390            b_V=b_V,
391            tan_gamma_V=tan_gamma_V,
392        )
393    else:
394        sig_y = horizontal_stddev(
395            x_safe,
396            scheme=scheme,
397            stability_class=stability_class,
398            tan_gamma_H=tan_gamma_H,
399            source_half_width=source_half_width,
400        )
401        sig_z = vertical_stddev(
402            x_safe,
403            scheme=scheme,
404            stability_class=stability_class,
405            tan_gamma_V=tan_gamma_V,
406        )
407
408    u = wind.speed[:, None]  # (T, 1)
409
410    exp_y = jnp.exp(-0.5 * (y_cross / sig_y) ** 2)
411
412    dz_t = jnp.broadcast_to(dz[None, :], (T, N))
413    h = source.z
414    H = mixing_height
415    exp_z_direct = jnp.exp(-0.5 * (dz_t / sig_z) ** 2)
416    exp_z_ground = jnp.exp(-0.5 * ((dz_t + 2 * h) / sig_z) ** 2)
417    exp_z_inversion = jnp.exp(-0.5 * ((dz_t - 2 * (H - h)) / sig_z) ** 2)
418    exp_z_ceiling2 = jnp.exp(-0.5 * ((dz_t + 2 * H) / sig_z) ** 2)
419    exp_z = exp_z_direct + exp_z_ground + exp_z_inversion + exp_z_ceiling2
420
421    coupling = downwind_mask / (2 * jnp.pi * u * sig_y * sig_z) * exp_y * exp_z
422    return methane_kg_m3_to_ppm(coupling)

 82        post_prec = T / sigma2 + 1.0 / sigma_bg2
 83        post_var = 1.0 / post_prec
 84        # posterior mean = post_var * (sum_t residuals[t,n]) / sigma^2
 85        post_mean = post_var * jnp.sum(residuals, axis=0) / sigma2
 86        return post_mean + jnp.sqrt(post_var) * jax.random.normal(key, shape=post_mean.shape)

 55        r"""Positive-definite Riemannian metric inverse `G^{-1}` at `x`.
 56
 57        Parameters
 58        ----------
 59        x : Array, shape (7,)
 60            Point at which to evaluate the metric.
 61        log_posterior_fn : Callable
 62            `(x) -> scalar` log-posterior, holding `background`/`sigma2` fixed.

 64        Returns
 65        -------

105    def _log_proposal(
106        self, x_prop: Array, x_curr: Array, grad: Array, sqrt_G: Array, eps: float
107    ) -> Array:
108        r"""Log density of the M-MALA proposal `q(x_prop | x_curr)`.
109
110        Parameters
111        ----------
112        x_prop : Array, shape (7,)
113            Proposed point.
114        x_curr : Array, shape (7,)
115            Current point.
116        grad : Array, shape (7,)
117            Gradient of the log-posterior at `x_curr`.
118        sqrt_G : Array, shape (7, 7)
119            `G^{-1/2}` at `x_curr` (from `sqrt_inv_hess`).
120        eps : float
121            Step size `zeta`.
122
123        Returns
124        -------
125        Array
126            Scalar log proposal density, up to an additive normalizing
127            constant (sufficient for the Metropolis-Hastings ratio).
128
129        Notes
130        -----
131        Paper Mapping: Newman et al. (2024), Eq. (10), §3.2.
132
133        .. math::
134            \theta^* \sim \mathcal{N}_n\!\left(
135                \theta^{(l-1)} + \tfrac{1}{2}\zeta^{(l-1)} G^{-1}
136                \nabla\log p(\theta^{(l-1)}\mid \mathbf{d}),\;
137                \zeta^{(l-1)} G^{-1}\right)
138
139        Note the implementation's `eps**2` scaling is an equivalent
140        reparametrization of the paper's `zeta` (step size in "standard
141        deviation" units rather than variance units).
142        """
143        mu = x_curr + 0.5 * eps**2 * (sqrt_G @ sqrt_G @ grad)
144        diff = x_prop - mu
145        G = jnp.linalg.inv(sqrt_G @ sqrt_G)
146        return -0.5 / eps**2 * diff @ G @ diff
